Required JSON format

[
  {
    "subject": "Pathophysiology",
    "system": "Cardiovascular System",
    "stem": "A 62-year-old man presents with crushing substernal chest pain...\n\nWhich of the following best explains this patient's hypotension?",
    "choices": ["Choice A", "Choice B", "Choice C", "Choice D", "Choice E"],
    "correct": "E",
    "objective": "Identify right ventricular infarction as a preload-dependent cause of hypotension after inferior MI.",
    "rationale": "Right ventricular infarction commonly occurs with inferior MI due to RCA occlusion...",
    "whyWrong": {
      "A": "Why A is wrong.",
      "B": "Why B is wrong.",
      "C": "Why C is wrong.",
      "D": "Why D is wrong."
    }
  }
